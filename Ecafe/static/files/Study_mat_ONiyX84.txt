  function initS3Upload(id, inputFiled) {
        let self = this;
        self.uploadedFiles = [];
        self.deletedFiles = [];
        self.uploader = new qq.s3.FineUploader({
            validation: {
                allowedExtensions: ['docx', 'pptx', 'xlsx', 'xls', 'doc', 'pdf', 'odt', 'txt', 'rtf', 'ppt','jpg','png'],
                sizeLimit: '524288000'
            },
            element: document.getElementById(id),
            request: {

                endpoint: 'https://amsfilestore.s3.amazonaws.com',
                accessKey: 'AKIAIZ3OJULHNA3S24IQ'
            },

            deleteFile: {
                enabled: false,
            },
            signature: {
                endpoint: '/libcommon/s3/endpoint-cors.php',
                version: 4
            },
            objectProperties: {
                region: 'ap-south-1',
                key: function (fileId) {
                    var filename = self.uploader.getName(fileId);
                    var uuid = self.uploader.getUuid(fileId);
                    var ext = filename.substr(filename.lastIndexOf('.') + 1);
                    return 'STJOSEPHS/work_register_notes/2022/' + uuid + '-10259' + '.' + ext;
                }
            },

            callbacks: {
                onSubmitted: function (id, name) {
                    self.uploadedFileName = name;
                },
                onUpload: function (id, name) {
                    $('#vue-app').block();
                },
                onError: function (id, name, errorReason, xhrOrXdr) {
                    $('#vue-app').unblock();
                    if (errorReason.toLowerCase().indexOf('policy expired') !== -1) {
                        $.simpleDialog({
                            title: "Incorrect System Time",
                            message: "Your system date and/or time is incorrect. Please correct it and click retry to upload the file."
                        });
                    }
                },

                onComplete: function (fileId, name, response) {
                    if (response.success) {
                        var fileInfo = {
                            name: name,
                            key: self.uploader.getKey(fileId),
                            bucket: self.uploader.getBucket(fileId)
                        };
                        jsonValue = JSON.stringify(fileInfo);
                        self.uploadedFiles.push(jsonValue);
                    }

                },
                onAllComplete: function () {
                    var uploads = self.uploader.getUploads({
                        status: qq.status.UPLOAD_SUCCESSFUL
                    });
                    if (uploads && uploads.length) {
                        $('.qq-upload-button').addClass('hidden');
                        SaveUploadedNote();

    function readCookie(name) {
        var cookiename = name + "=";
        var ca = document.cookie.split(';');
        for (var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
            if (c.indexOf(cookiename) == 0) return c.substring(cookiename.length, c.length);
        }
        return null;
    }
